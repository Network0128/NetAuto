스위치를 파일로 생성: myswitches
sh vlan br   ← vlan 정보 조회

import telnetlib

#HOST = "10.1.1.11"
username = "ccnp"
password = "cisco"

f = open("myswitches")

for IP in f:
    IP = IP.strip()
    print("Switch's IP address: ", IP)
    tn = telnetlib.Telnet(IP)
    tn.read_until(b"Username: ")
    tn.write(username.encode('ascii') + b"\n")
    tn.read_until(b"Password: ")
    tn.write(password.encode('ascii') + b"\n")
    tn.write(b"conf t\n")

    for n in range(2,21):
        tn.write(b"vlan " + str(n).encode('ascii') + b"\n")
        tn.write(b"name Python_VLAN_" + str(n).encode('ascii') + b"\n")

    tn.write(b"end\n")
    tn.write(b"terminal length 0\n")
    tn.write(b"sh vlan br\n")
    tn.write(b"exit\n")
    print(tn.read_all().decode('ascii'))



라우터 목록을 파일로 생성: myrouters
sh ip int br   ← 인터페이스 IP 정보 조회

import telnetlib

#HOST = "10.1.1.11"
username = "ccnp"
password = "cisco"

f = open("myrouters")

for IP in f:
    IP = IP.strip()
    print("Router's IP address: ", IP)
    tn = telnetlib.Telnet(IP)
    tn.read_until(b"Username: ")
    tn.write(username.encode('ascii') + b"\n")
    tn.read_until(b"Password: ")
    tn.write(password.encode('ascii') + b"\n")
    tn.write(b"conf t\n")
    tn.write(b"int l0\n")      ← 삭제 시 tn.write(b"no int l0\n")
    tn.write(b"ip add 1.1.1.1 255.255.255.0\n")
    tn.write(b"end\n")
    #tn.write(b"terminal length 0\n")
    tn.write(b"sh ip int br\n")
    tn.write(b"exit\n")
    print(tn.read_all().decode('ascii'))
